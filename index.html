function initSnowfall() { for (let i = 0; i < 100; i++) { snowflakes.push(createSnowflake()); } drawSnowfall(); } function createSnowflake() { return { x: Math.random() * canvas.width, y: 0, size: Math.random() * 5 + 2, speed: Math.random() * 3 + 1, }; } function drawSnowflake(snowflake) { ctx.beginPath(); ctx.arc(snowflake.x, snowflake.y, snowflake.size, 0, Math.PI * 2); ctx.fillStyle = '#fff'; ctx.fill(); } function updateSnowflake(snowflake) { snowflake.y += snowflake.speed; if (snowflake.y > canvas.height) { snowflake.y = 0; snowflake.x = Math.random() * canvas.width; } }
